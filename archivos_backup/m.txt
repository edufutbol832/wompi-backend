  <script>
    function mostrarModulo(modulo) {
      document.querySelectorAll('.menu-principal, .modulo').forEach(div => div.classList.remove('visible'));

      switch (modulo) {
        case 'instructivo':
          document.getElementById('moduloInstructivo').classList.add('visible');
          break;
        case 'simulacion':
          document.getElementById('moduloSimulacion').classList.add('visible');
          break;
        case 'lista':
          document.getElementById('moduloLista').classList.add('visible');
          break;
        case 'pago':
          document.getElementById('moduloPago').classList.add('visible');
          cargarBotonWompi();
          break;
        case 'suscripcion':
          document.getElementById('moduloSuscripcion').classList.add('visible');
          break;
      }
    }

    function volverAlMenu() {
      document.querySelectorAll('.modulo').forEach(div => div.classList.remove('visible'));
      document.getElementById('menuPrincipal').classList.add('visible');
      document.getElementById('wompi-container').innerHTML = '<p>Generando referencia y firma...</p>';
    }

    function cargarBotonWompi() {
      const referencia = "edupago-" + Date.now();
      const datos = {
        amount_in_cents: 150000,
        currency: "COP",
        reference: referencia
      };

      fetch("https://wompi-backend-zodk.onrender.com/generate-signature", {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify(datos)
      })
      .then(response => response.json())
      .then(data => {
        if (!data.signature) throw new Error("Firma no recibida");

        const script = document.createElement("script");
        script.src = "https://checkout.wompi.co/widget.js";
        script.setAttribute("data-render", "button");
        script.setAttribute("data-public-key", "pub_test_K9UBxtRMrdrjaNBNdpuqoLYxmh8feExG");
        script.setAttribute("data-currency", "COP");
        script.setAttribute("data-amount-in-cents", "150000");
        script.setAttribute("data-reference", referencia);
        script.setAttribute("data-signature:integrity", data.signature);
        script.setAttribute("data-redirect-url", `https://wompi-backend-zodk.onrender.com/respuesta.html?id=${referencia}`);

        const container = document.getElementById("wompi-container");
        container.innerHTML = "";
        container.appendChild(script);
      })
      .catch(error => {
        document.getElementById("wompi-container").innerHTML = `<p style="color:red;">âŒ Error al generar firma.</p><p>${error.message}</p>`;
        console.error("Error:", error);
      });
    }
  </script>
